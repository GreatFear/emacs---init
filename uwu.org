  (defun create-note-type-menu ()
    (let ((buf (get-buffer-create "*menu*")))
      (with-current-buffer buf
        (erase-buffer)
        (insert "0 - Fleeting Notes\n")
        (insert "1 - Reference Notes\n")
        (insert "2 - Permanent Notes\n")
        (insert "3 - World Building\n")
        )
      (display-buffer buf)))

  (defun response (input)
    (pcase input
      ("2" (completing-read "Worlds: " '("0" "1") nil t))
      ("4" (completing-read "Types of Notes: " '("0" "1" "2" "3") nil t))
      ("3" (completing-read "Types of Notes: " '("0" "1" "2") nil t))
      ("7" (completing-read "World Building Topcs: " '("0" "1" "2" "3" "4" "5" "6") nil t))
      )
    )

  (defun murder-menu (input)
    "Kills whatever is inputted"
    (kill-buffer input))

  (defun menu ()
    (prog2
        (create-note-type-menu)
        (response "4")
      (murder-menu "*menu*")))
#+END_SRC

***** World Building Notes
#+BEGIN_SRC emacs-lisp
  (defun world-menu ()
    "Display the world" 
    (let ((buf (get-buffer-create "*world-menu*")))
      (with-current-buffer buf
        (erase-buffer)
        (insert "0 - The Great Forest\n")
        (insert "1 - The Rotting Beserker\n")
        )
      (display-buffer buf)))

  (defun topic-menu ()
    "Display the topic"
    (let ((buf (get-buffer-create "*topic-menu*")))
      (with-current-buffer buf
        (erase-buffer)
        (insert "0 - Map of Content\n")
        (insert "1 - Concept\n")
        (insert "2 - Event")
        (insert "3 - Region\n")
        (insert "4 - Nation\n")
        (insert "5 - Race\n")
        (insert "6 - Character\n")
        )
      (display-buffer buf)))

  (defun realm-menu ()
    "Display the realm" 
    (let ((buf (get-buffer-create "*realm-menu*")))
      (with-current-buffer buf
        (erase-buffer)
        (insert "0 - Lower Realm\n")
        (insert "1 - Middle Realm\n")
        (insert "2 - Upper Realm")
        )
      (display-buffer buf)))

  (defun choosing-world ()
    (prog2
        (world-menu)
        (response "2")
      (murder-menu "*world-menu*")))

  (defun choosing-topic ()
    (prog2
        (topic-menu)
        (response "7")
      (murder-menu "*topic-menu*")))

  (defun choosing-realm (input)
    (prog2
        (realm-menu)
        (response "3")
      (murder-menu "*realm-menu*")))

  (defun deciding-realm (world input)
    (if (string= world "TheGreatForest") 
        (pcase input
      	("0" "Lower")
      	("1" "Middle")
      	("2" "Upper")
      	)
      )
    (string= world "TheGreatForest")
    )

  (defun response-world (input)
    (pcase input
      ("0" "TheGreatForest")
      ("1" "TheRottingBeserker")
      ))

  (defun response-topic (input)
    (pcase input
      ("0" "MOC")
      ("1" "Concept")
      ("2" "Event")
      ("3" "Region")
      ("4" "Nation")
      ("5" "Race")
      ("6" "Character")
      ))

  (defun with-or-without-realm (world realm topic)
    (if (string= world "TheGreatForest")
        (format "#+title: ${title}\n#+filetags: :%s:%s:%s:\nPrevious Link: \n\n**** Permenant Links\n" world realm topic)
      (format "#+title: ${title}\n#+filetags: :%s:%s:\nPrevious Link: \n\n**** Permenant Links\n" world topic))
    )

  (defun creation-of-world-building-notes ()
    (let* ((world (choosing-world))
      	 (realm (deciding-realm (choosing-realm)))
      	 (topic (choosing-topic))
      	 (final (with-or-without-realm world realm topic))
      	 )
      (setq org-roam-capture-templates ()
      	  ("m" "Map of Contents" plain
      	   final
      	   :if-new (file "3-world-building/%<%Y%m%d%H%M%S>-${slug}.org")
      	   :unnarrowed t))
      )
    )

#+END_SRC
